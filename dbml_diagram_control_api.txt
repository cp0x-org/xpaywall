// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table users {
  id uuid [primary key]

  email varchar [not null, unique]
  username varchar

  created_at timestamp
  updated_at timestamp
}

Table projects {
  id uuid [primary key]

  owner_user_id uuid [not null]

  name varchar [not null]
  slug varchar [not null, unique]

  created_at timestamp
  updated_at timestamp
}

Table project_members {
  id uuid [primary key]

  project_id uuid [not null]
  user_id uuid [not null]

  role varchar [not null, note: 'owner | admin | viewer']

  created_at timestamp

  indexes {
    (project_id, user_id) [unique]
  }
}

//
// PAYMENT STACK
//

Table payment_channels {
  id uuid [primary key]

  protocol varchar [not null, note: 'x402 | mpp']
  network varchar [not null, note: 'base | solana | ethereum']

  scheme varchar [not null, note: 'charge | session | upto']

  enabled boolean [not null, default: true]

  metadata jsonb [note: 'Future protocol-specific settings']

  created_at timestamp
  updated_at timestamp

  indexes {
    (protocol, network, scheme) [unique]
  }
}

Table payment_channel_assets {
  id uuid [primary key]

  payment_channel_id uuid [not null]

  asset_symbol varchar [not null, note: 'USDC | USDT | CUSTOM']

  asset_address varchar [note: 'Token contract address']
  decimals integer

  metadata jsonb

  created_at timestamp
  updated_at timestamp

  indexes {
    (payment_channel_id, asset_symbol) [unique]
  }
}

//
// PROJECT PAYMENT CONFIG
//

Table project_payment_configs {
  id uuid [primary key]

  project_id uuid [not null]

  payment_channel_id uuid [not null]

  payment_channel_asset_id uuid [note: 'nullable for non-token methods']

  name varchar [not null]

  //
  // WHERE TO RECEIVE MONEY
  //

  payout_address varchar [note: 'Wallet address / merchant account']

  enabled boolean [not null, default: true]

  created_at timestamp
  updated_at timestamp
}


//
// OUTBOUND
//

Table outbound_targets {
  id uuid [primary key]

  project_id uuid [not null]

  name varchar [not null]

  base_url varchar [not null, note: 'https://api.example.com']

  auth_header_name varchar
  auth_header_value varchar

  is_default boolean [not null, default: false]

  created_at timestamp
  updated_at timestamp
}

Table outbound_routes {
  id uuid [primary key]

  project_id uuid [not null]

  outbound_target_id uuid [not null]

  name varchar [not null]

  path_pattern varchar [not null, note: '/weather/*']
  price_amount integer [not null]

  created_at timestamp
  updated_at timestamp
}

//
// INBOUND
//

Table inbound_routes {
  id uuid [primary key]

  project_id uuid [not null]

  outbound_route_id uuid [not null]

  name varchar [not null]

  methods varchar [note: 'GET,POST']

  path_pattern varchar [not null, note: '/project-slug/weather/*']

  match_type varchar [not null, note: 'exact | glob | prefix']

  priority integer [not null, default: 0]

  enabled boolean [not null, default: true]

  created_at timestamp
  updated_at timestamp

  indexes {
    path_pattern [unique]
  }
}

//
// OPTIONAL ROUTE-SPECIFIC PAYMENT OVERRIDES
//

Table route_payment_overrides {
  id uuid [primary key]

  inbound_route_id uuid [not null]

  project_payment_config_id uuid [not null]

  price_amount integer [not null]

  enabled boolean [not null, default: true]

  created_at timestamp
  updated_at timestamp

  indexes {
    (inbound_route_id, project_payment_config_id) [unique]
  }
}

Ref: project_members.user_id > users.id

Ref: outbound_targets.project_id > projects.id

Ref: route_payment_overrides.inbound_route_id > inbound_routes.id
